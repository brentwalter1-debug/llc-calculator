    <script>
        const formatter = new Intl.NumberFormat('en-US', {
            style: 'currency',
            currency: 'USD',
        });

        function animateValue(obj, start, end, duration) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                const easeOutProgress = 1 - Math.pow(1 - progress, 3);
                const currentVal = (easeOutProgress * (end - start) + start);
                obj.innerHTML = formatter.format(currentVal);
                if (progress < 1) {
                    window.requestAnimationFrame(step);
                }
            };
            window.requestAnimationFrame(step);
        }

        function calculateTax() {
            const incomeInput = document.getElementById('monthlyIncome').value;
            const monthlyIncome = parseFloat(incomeInput);

            const expensesInput = document.getElementById('monthlyExpenses').value;
            const monthlyExpenses = expensesInput ? parseFloat(expensesInput) : 0;

            if (isNaN(monthlyIncome) || monthlyIncome < 0 || monthlyExpenses < 0) {
                alert("Please enter valid, positive numbers.");
                return;
            }

            const netMonthlyProfit = monthlyIncome - monthlyExpenses;
            
            if (netMonthlyProfit <= 0) {
                document.getElementById('resultArea').style.display = 'block';
                document.getElementById('taxAmount').innerHTML = "$0.00";
                // Reset bars
                document.getElementById('takeHomeBar').style.width = '0%';
                document.getElementById('taxBar').style.width = '0%';
                document.getElementById('thresholdMessage').style.display = 'none';
                document.getElementById('ctaBtn').style.display = 'none';
                return;
            }

            const annualProfit = netMonthlyProfit * 12;
            const selfEmploymentTaxRate = 0.153; 
            const estimatedTax = annualProfit * selfEmploymentTaxRate;
            const takeHomePay = annualProfit - estimatedTax;

            document.getElementById('resultArea').style.display = 'block';
            
            const taxAmountElement = document.getElementById('taxAmount');
            animateValue(taxAmountElement, 0, estimatedTax, 1200);

            // Calculate percentages for the visual bar
            const takeHomePercentage = (takeHomePay / annualProfit) * 100;
            const taxPercentage = (estimatedTax / annualProfit) * 100;

            // Small delay so the bar animates smoothly after the area is revealed
            setTimeout(() => {
                document.getElementById('takeHomeBar').style.width = `${takeHomePercentage}%`;
                document.getElementById('taxBar').style.width = `${taxPercentage}%`;
            }, 50);

            const thresholdMessage = document.getElementById('thresholdMessage');
            const ctaBtn = document.getElementById('ctaBtn');

            if (netMonthlyProfit > 4000) {
                setTimeout(() => {
                    thresholdMessage.style.display = 'block';
                    ctaBtn.style.display = 'inline-block';
                }, 800);
            } else {
                thresholdMessage.style.display = 'none';
                ctaBtn.style.display = 'none';
            }
        }
    </script>
