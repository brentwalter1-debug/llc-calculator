    <script>
        const formatter = new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD' });

        // Dictionary of estimated State Tax Rates (simplified for MVP)
        const stateTaxRates = {
            "none": 0.0,
            "CA": 0.08,
            "NY": 0.06,
            "IL": 0.0495
        };

        // --- LOCAL STORAGE FUNCTIONALITY ---
        // Saves data as the user types
        function saveData() {
            localStorage.setItem("app_income", document.getElementById('monthlyIncome').value);
            localStorage.setItem("app_expenses", document.getElementById('monthlyExpenses').value);
            localStorage.setItem("app_state", document.getElementById('userState').value);
        }

        // Loads data when the page opens
        window.onload = function() {
            if(localStorage.getItem("app_income")) document.getElementById('monthlyIncome').value = localStorage.getItem("app_income");
            if(localStorage.getItem("app_expenses")) document.getElementById('monthlyExpenses').value = localStorage.getItem("app_expenses");
            if(localStorage.getItem("app_state")) document.getElementById('userState').value = localStorage.getItem("app_state");
        }

        // --- ANIMATION FUNCTION ---
        function animateValue(obj, start, end, duration) {
            let startTimestamp = null;
            const step = (timestamp) => {
                if (!startTimestamp) startTimestamp = timestamp;
                const progress = Math.min((timestamp - startTimestamp) / duration, 1);
                const easeOut = 1 - Math.pow(1 - progress, 3);
                obj.innerHTML = formatter.format(easeOut * (end - start) + start);
                if (progress < 1) window.requestAnimationFrame(step);
            };
            window.requestAnimationFrame(step);
        }

        // --- STEP 1: TRIGGER LEAD GATE ---
        function triggerLeadGate() {
            const income = parseFloat(document.getElementById('monthlyIncome').value);
            if (isNaN(income) || income <= 0) {
                alert("Please enter your estimated revenue first.");
                return;
            }
            // Hide the calculate button and show the email form
            document.getElementById('initCalcBtn').style.display = 'none';
            document.getElementById('leadGate').style.display = 'block';
        }

        // --- STEP 2: UNLOCK RESULTS & CALCULATE ---
        function unlockResults() {
            const email = document.getElementById('userEmail').value;
            if(!email.includes('@')) {
                alert("Please enter a valid email to unlock your strategy.");
                return;
            }

            // In the future, we will write code here to send the email to your database!

            document.getElementById('leadGate').style.display = 'none';
            document.getElementById('resultArea').style.display = 'block';

            // Core Math
            const income = parseFloat(document.getElementById('monthlyIncome').value);
            const expenses = parseFloat(document.getElementById('monthlyExpenses').value) || 0;
            const state = document.getElementById('userState').value;
            
            const netMonthlyProfit = income - expenses;
            const annualProfit = Math.max(0, netMonthlyProfit * 12);
            
            const fedTaxRate = 0.153; 
            const stateTaxRate = stateTaxRates[state];
            
            const estimatedFedTax = annualProfit * fedTaxRate;
            const estimatedStateTax = annualProfit * stateTaxRate;
            const totalTax = estimatedFedTax + estimatedStateTax;
            const takeHomePay = annualProfit - totalTax;

            // DOM Updates
            animateValue(document.getElementById('taxAmount'), 0, totalTax, 1200);
            
            if(estimatedStateTax > 0) {
                document.getElementById('stateTaxSub').innerHTML = `Includes ${formatter.format(estimatedStateTax)} in Estimated State Tax`;
            } else {
                document.getElementById('stateTaxSub').innerHTML = `You live in a great state! No state income tax applied.`;
            }

            // Visual Bar Logic
            setTimeout(() => {
                const takeHomePct = annualProfit > 0 ? (takeHomePay / annualProfit) * 100 : 0;
                const taxPct = annualProfit > 0 ? (totalTax / annualProfit) * 100 : 0;
                document.getElementById('takeHomeBar').style.width = `${takeHomePct}%`;
                document.getElementById('taxBar').style.width = `${taxPct}%`;
            }, 100);
        }

        // --- PDF GENERATION FUNCTION ---
        function downloadPDF() {
            // Selects just the results area to turn into a document
            const element = document.getElementById('pdfContent');
            const opt = {
                margin:       0.5,
                filename:     'My_LLC_Tax_Strategy.pdf',
                image:        { type: 'jpeg', quality: 0.98 },
                html2canvas:  { scale: 2 },
                jsPDF:        { unit: 'in', format: 'letter', orientation: 'portrait' }
            };
            // This runs the imported library
            html2pdf().set(opt).from(element).save();
        }
    </script>
